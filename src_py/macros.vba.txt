Sub MacroEliminarColumnasPorNombre()

Dim tabla As ListObject
Dim colum As Variant
Dim i As Integer
Dim totColum As Integer

Set tabla = ActiveSheet.ListObjects(1)

'Borrar primeras columnas
For Each colum In Array("Hora de inicio", "Hora de finalización", "Correo electrónico", "Nombre", "Total de puntos", "Comentarios del cuestionario", "Hora de publicación de la calificación", "Nombre completo", "Puntos: Nombre completo", "Comentarios: Nombre completo", "Correo electrónico1")
    tabla.ListColumns(colum).Delete
Next colum

'Borrar columnas que tienen la palabra Comentarios o Puntos
totColum = tabla.ListColumns.Count
For i = totColum To 1 Step -1 'Disminuye para que elimine de izquierda a derecha y no haya problemas al recorrerse las columnas
    'InStr para buscar un texto dentro de otro -> empieza en 1 porque representa que busca desde el primer caracter-> devuelve 0 si NO encuentra el texto, >0 si SÍ encuentra el texto
    If InStr(1, tabla.ListColumns(i).Name, "Comentarios:", vbBinaryCompare) > 0 Or InStr(1, tabla.ListColumns(i).Name, "Puntos:", vbBinaryCompare) > 0 Then
        tabla.ListColumns(i).Delete
    End If
Next i

End Sub


Sub CopiarTabla()

Dim tablaCopiar As ListObject
Dim hojaCopiar As Worksheet

'inicializar variables
Set tablaCopiar = ActiveSheet.ListObjects(1)
Set hojaCopiar = Sheets.Add(After:=Sheets("Sheet1")) 'agregar "hojaCopiar" al libro
hojaCopiar.Name = "HojaPolars" 'ponerle nombre a la hoja

'Copiar la hoja
tablaCopiar.Range.Copy Destination:=hojaCopiar.Range("A1")

End Sub
